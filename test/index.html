ARCH models

a0 <- 0.2
a1 <- 0.4
n <- 1000L
ee <- rnorm(n+1)
zt <- numeric(length = n+1)
zt[1] <- ee[1]
for (i in 2:(n+1)) {
  ht <- a0 + a1*zt[i-1]^2
  zt[i] <- ee[i]*sqrt(ht)
}
plot(zt,type="l")

acf(zt^2)
